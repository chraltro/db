1-- Fetch active users with recent activity
2SELECT
3 u.id,
4 u.name,
5 u.email,
6 COUNT(s.id) AS session_count,
7 MAX(s.created_at) AS last_seen
8FROM users u
9LEFT JOIN sessions s ON s.user_id = u.id
10WHERE u.status = 'active'
11 AND s.created_at > NOW() - INTERVAL '30 days'
12GROUP BY u.id, u.name, u.email
13HAVING COUNT(s.id) > 5
14ORDER BY session_count DESC
15LIMIT 25;
